const debounce = (func) => {
  let timeId = null
  return {
    if(timeId){
      clearTimeout(timeId)
    }
    timeId = setTimeout(()=>{
      func()
    },500}
  }
}

const keyUpEvent = ev => {
  const textInput = ev.target.value
  debounce(this.setState({textInput})
}
